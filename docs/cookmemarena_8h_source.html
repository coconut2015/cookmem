74  FixedArena(void* ptr, std::size_t size)
75  : m_page (), m_size (), m_used (false)
76  {
77  std::size_t remain = (std::size_t)ptr & 0xfUL;
78  if (remain)
79  {
80  std::size_t inc = (0x10 - remain);
81  m_page = (void*)((char*)ptr + inc);
82  size -= inc;
83  }
84  else
85  {
86  m_page = ptr;
87  }
88  size &= 0xfffffffffffffff0UL;
89  m_size = size;
90  }

100  void* getSegment(std::size_t& size)
101  {
102  if (m_used || size > m_size)
103  {
104  return NULL;
105  }
106  m_used = true;
107  size = m_size;
108  return m_page;
109  }

121  {
122  if (!m_used || ptr != m_page || size != m_size)
123  {
124  return true;
125  }
126  m_used = false;
127  return false;
128  }
