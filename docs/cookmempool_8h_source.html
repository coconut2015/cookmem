45 /* assign tree index for size S to variable I. Use x86 asm if possible */
46 #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
47 #define cookmem_getLargeBinIndex(S, I)\
48 {\
49  unsigned int X = S >> TREEBIN_SHIFT;\
50  if (X == 0)\
51  I = 0;\
52  else if (X > 0xFFFF)\
53  I = NTREEBINS-1;\
54  else {\
55  unsigned int K = (unsigned) sizeof(X)*__CHAR_BIT__ - 1 - (unsigned) __builtin_clz(X); \
56  I = (BinIndexType)((K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1)));\
57  }\
58 }
59 
60 #elif defined (__INTEL_COMPILER)
61 #define cookmem_getLargeBinIndex(S, I)\
62 {\
63  std::size_t X = S >> TREEBIN_SHIFT;\
64  if (X == 0)\
65  I = 0;\
66  else if (X > 0xFFFF)\
67  I = NTREEBINS-1;\
68  else {\
69  unsigned int K = _bit_scan_reverse (X); \
70  I = (BinIndexType)((K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1)));\
71  }\
72 }
73 
74 #elif defined(_MSC_VER) && _MSC_VER>=1300
75 #define cookmem_getLargeBinIndex(S, I)\
76 {\
77  std::size_t X = S >> TREEBIN_SHIFT;\
78  if (X == 0)\
79  I = 0;\
80  else if (X > 0xFFFF)\
81  I = NTREEBINS-1;\
82  else {\
83  unsigned int K;\
84  _BitScanReverse((DWORD *) &K, (DWORD) X);\
85  I = (BinIndexType)((K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1)));\
86  }\
87 }
88 
89 #else /* GNUC */
90 #define cookmem_getLargeBinIndex(S, I)\
91 {\
92  std::size_t X = S >> TREEBIN_SHIFT;\
93  if (X == 0)\
94  I = 0;\
95  else if (X > 0xFFFF)\
96  I = NTREEBINS-1;\
97  else {\
98  unsigned int Y = (unsigned int)X;\
99  unsigned int N = ((Y - 0x100) >> 16) & 8;\
100  unsigned int K = (((Y <<= N) - 0x1000) >> 16) & 4;\
101  N += K;\
102  N += K = (((Y <<= K) - 0x4000) >> 16) & 2;\
103  K = 14 - N + ((Y <<= K) >> 15);\
104  I = (K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1));\
105  }\
106 }
107 #endif /* GNUC */
108 
109 #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
110 #define cookmem_bit2treeIndex(X, I)\
111 {\
112  unsigned int J;\
113  J = __builtin_ctz(X); \
114  I = (BinIndexType)J;\
115 }
116 
117 #elif defined (__INTEL_COMPILER)
118 #define cookmem_bit2treeIndex(X, I)\
119 {\
120  unsigned int J;\
121  J = _bit_scan_forward (X); \
122  I = (BinIndexType)J;\
123 }
124 
125 #elif defined(_MSC_VER) && _MSC_VER>=1300
126 #define cookmem_bit2treeIndex(X, I)\
127 {\
128  unsigned int J;\
129  _BitScanForward((DWORD *) &J, X);\
130  I = (BinIndexType)J;\
131 }
132 
133 #elif USE_BUILTIN_FFS
134 #define cookmem_bit2treeIndex(X, I) I = ffs(X)-1
135 
136 #else
137 #define cookmem_bit2treeIndex(X, I)\
138 {\
139  unsigned int Y = X - 1;\
140  unsigned int K = Y >> (16-4) & 16;\
141  unsigned int N = K; Y >>= K;\
142  N += K = Y >> (8-3) & 8; Y >>= K;\
143  N += K = Y >> (4-2) & 4; Y >>= K;\
144  N += K = Y >> (2-1) & 2; Y >>= K;\
145  N += K = Y >> (1-0) & 1; Y >>= K;\
146  I = (BinIndexType)(N + Y);\
147 }
148 #endif /* GNUC */

185  class MemChunk
186  {
187  public:
192  static const size_type BIT_MASK = ((size_t)-1) ^ 0x0f;
193  static const size_type BIT_USED = 1;
194  static const size_type BIT_NOTEXACTSIZE = 2;
195 
196  private:
197  size_type m_prevFootSize; /* Size of previous chunk (if free). */
198  size_type m_size; /* Size */
199 
200  public:
201  inline size_type
202  getChunkSize () const
203  {
204  return m_size & BIT_MASK;
205  }
206 
207  inline void
208  setFreeChunkSize (size_type chunkSize)
209  {
210  m_size = chunkSize;
211  ((MemChunk*)((char*)(this) + (chunkSize)))->m_prevFootSize = chunkSize;
212  }
213 
214  inline bool
215  isUsed () const
216  {
217  return m_size & BIT_USED;
218  }
219 
220  inline size_type
221  getUserSize () const
222  {
223  size_type chunkSize = getChunkSize ();
224  if (m_size & BIT_NOTEXACTSIZE)
225  {
226  char diff = *((char*)(this) + chunkSize - 1);
227  if (diff <= 0)
228  {
229  throw Exception (MEM_ERROR_PADDING, "Invalid padding byte value");
230  }
231  return chunkSize - CHUNK_OVERHEAD - diff;
232  }
233  return chunkSize - CHUNK_OVERHEAD;
234  }
235 
236  inline void
237  setUsedSize (bool storingExactSize, size_type userSize)
238  {
239  size_type chunkSize = getChunkSize ();
240  m_size |= BIT_USED;
241  if (storingExactSize)
242  {
243  userSize += CHUNK_OVERHEAD;
244  if (userSize != chunkSize)
245  {
246  m_size |= BIT_NOTEXACTSIZE;
247  char diff = (char)(chunkSize - userSize);
248  *(((char*)this) + chunkSize - 1) = diff;
249  }
250  }
251  ((MemChunk*)((char*) (this) + (chunkSize)))->m_prevFootSize = chunkSize | BIT_USED;
252  }
253  };

530  T*
531  allocate (size_type userSize)
532  {
533  size_type chunkSize;
534 
535  size_type allocSize = getMinAllocSize (userSize);
536 
537  if (allocSize < MIN_LARGE_REQUEST)
538  {
539  BinIndexType binIndex;
540  BinIndexType smallBits;
541  chunkSize = (allocSize < MIN_REQUEST) ? MIN_CHUNK_SIZE : calcChunkSize (allocSize);
542  binIndex = getSmallBinIndex (chunkSize);
543  smallBits = m_smallMap >> binIndex;
544 
545  /*
546  * Check if two smallest chunk lists that can satisfy the request.
547  * Note that since the chunk is quite tight, it is not necessary
548  * to split the chunk since there will not be enough rooms.
549  */
550  if ((smallBits & 0x3U) != 0)
551  {
552  MemChunk* chunk;
553  binIndex += ~smallBits & 1; /* Uses next bin if idx empty */
554 
555  chunk = removeSmallChunkList (binIndex);
556  COOKMEM_ASSERT(chunk != nullptr && chunk->getChunkSize() == getSmallBinSize (binIndex));
557 
558  setUsed (chunk, userSize);
559 
560  return getUserPointer (chunk, userSize);
561  }
562  else
563  {
564  if (smallBits != 0)
565  {
566  /* Use chunk in next nonempty smallbin */
567  MemChunk* chunk;
568  BinIndexType i;
569  BinIndexType leftbits = (smallBits << binIndex) & left_bits(idx2bit(binIndex));
570  BinIndexType leastbit = least_bit(leftbits);
571  cookmem_bit2treeIndex(leastbit, i);
572 
573  chunk = removeSmallChunkList(i);
574  COOKMEM_ASSERT(chunk != nullptr && chunk->getChunkSize() == getSmallBinSize(i));
575 
576  chunk = splitChunk (chunk, chunkSize);
577 
578  setUsed (chunk, userSize);
579 
580  return getUserPointer (chunk, userSize);
581  }
582  }
583  }
584  else if (allocSize >= MAX_REQUEST)
585  {
586  // The request size is too big.
587  m_logger.logAllocation (nullptr, userSize);
588  return nullptr;
589  }
590  else
591  {
592  chunkSize = calcChunkSize (allocSize);
593  }
594 
595  MemChunk* chunk;
596 
597  /*
598  * Check if we have enough space in existing chunks.
599  */
600  if (m_treeMap != 0)
601  {
602  chunk = treeMalloc (chunkSize);
603  }
604  else
605  {
606  chunk = nullptr;
607  }
608 
609  /*
610  * At this point, we have to allocate memory from arena.
611  */
612  if (chunk == nullptr)
613  {
614  chunk = arenaAlloc (chunkSize);
615 
616  if (chunk == nullptr)
617  {
618  m_logger.logAllocation (nullptr, userSize);
619  return nullptr;
620  }
621  }
622 
623  setUsed (chunk, userSize);
624 
625  return getUserPointer (chunk, userSize);
626  }

639  T*
640  reallocate (T* ptr, size_type newUserSize)
641  {
642  if (ptr == nullptr)
643  {
644  return allocate(newUserSize);
645  }
646 
647  MemChunk* chunk = mem2Chunk (ptr);
648  size_type oldAllocSize = chunk->getUserSize ();
649  size_type oldChunkSize = calcChunkSize(oldAllocSize);
650 
651  size_type newAllocSize = newUserSize;
652  size_type newChunkSize = (newAllocSize < MIN_REQUEST) ? MIN_CHUNK_SIZE : calcChunkSize(newAllocSize);
653 
654  if (oldChunkSize < newChunkSize)
655  {
656  // For now we use a simple algorithm that allocate new memory.
657  T* newPtr = allocate (newUserSize);
658  if (newPtr)
659  {
660  memcpy(newPtr, ptr, oldAllocSize);
661  deallocate (ptr);
662  }
663  return newPtr;
664  }
665  else
666  {
667  m_logger.logReallocation(ptr, oldAllocSize, newUserSize);
668 
669  // simply split the chunk
670  splitChunk (chunk, newChunkSize);
671  setUsed (chunk, newUserSize);
672  return ptr;
673  }
674  }

705  void
706  deallocate (T* ptr, size_type size = 0)
707  {
708  if (ptr != nullptr)
709  {
710  MemChunk* chunk = mem2Chunk (ptr);
711  if (!chunk->isUsed())
712  {
713  throw Exception (MEM_ERROR_DOUBLE_FREE, "potentially freeing an unused pointer");
714  }
715  size_type chunkSize = chunk->getChunkSize ();
716  if (m_padding)
717  {
718  COOKMEM_ASSERT (chunkSize > CHUNK_OVERHEAD);
719  size_type userSize = chunk->getUserSize () + CHUNK_OVERHEAD;
720  COOKMEM_ASSERT (userSize < chunkSize);
721  char diff = (char)(chunkSize - userSize);
722  COOKMEM_ASSERT ((diff > 0) && ((userSize + diff) == chunkSize));
723  COOKMEM_ASSERT (*(((char*)chunk) + chunkSize - 1) == diff);
724 
725  char* paddingPtr = ((char*)chunk) + userSize;
726  --diff;
727  if (diff > 8)
728  {
729  diff = 8;
730  }
731  for (int i = 0; i < diff; ++i)
732  {
733  if (paddingPtr[i] != m_paddingByte)
734  {
735  throw Exception (MEM_ERROR_PADDING, "padding byte got modified.");
736  }
737  }
738  }
739  m_logger.logDeallocation (ptr, chunk->getUserSize ());
740 
741  chunk->setFreeChunkSize (chunkSize);
742  addChunk (chunk);
743  }
744  else
745  {
746  m_logger.logDeallocation (ptr, size);
747  }
748  }

764  contains (T* ptr, bool checkUsed = false)
765  {
766  MemSegment* seg = m_segList;

771  {
772  if (checkUsed)

776  }

780  }
781  return false;

993  {

1028  inline MemChunk*
1029  splitChunk (MemChunk* chunk, size_type size)
1030  {
1031  // the size of the 2nd half after the split
1032  size_type remainSize = chunk->getChunkSize () - size;
1033 
1034  if (remainSize >= MIN_CHUNK_SIZE)
1035  {
1036  MemChunk* remainChunk = (MemChunk*)((char*)chunk + size);
1037 
1038  chunk->setFreeChunkSize (size);
1039 
1040  remainChunk->setFreeChunkSize (remainSize);
1041 
1042  addChunk (remainChunk);
1043  }
1044  return chunk;

1055  MemChunk*
1056  arenaAlloc (size_type chunkSize)
1057  {
1058  size_type estSize; /* allocation size */
1059 
1060  estSize = chunkSize + SEGMENT_OVERHEAD;
1061  if (estSize <= chunkSize)
1062  {
1063  // Technically can never reach here due to the MAX_REQUEST value
1064  //
1065  // However it is always good to have a check.
1066  return nullptr;
1067  }
1068 
1069  /*
1070  * Check user specified foot print limit
1071  */
1072  if (m_footprintLimit != 0)
1073  {
1074  size_type fp = m_footprint + estSize;
1075  if (fp <= m_footprint || fp > m_footprintLimit)
1076  {
1077  return nullptr;
1078  }
1079  }
1080 
1081  /*
1082  * Request memory segments from arena.
1083  */
1084  size_type segSize = estSize;
1085  MemSegment* seg = (MemSegment*)m_arena.getSegment (segSize);
1086  m_logger.logGetSegment (seg, segSize);
1087 
1088  /*
1089  * Initialize the segment obtained.
1090  */
1091  if (seg)
