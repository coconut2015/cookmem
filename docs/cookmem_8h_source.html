42 /* assign tree index for size S to variable I. Use x86 asm if possible */
43 #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
44 #define cookmem_getLargeBinIndex(S, I)\
45 {\
46  unsigned int X = S >> TREEBIN_SHIFT;\
47  if (X == 0)\
48  I = 0;\
49  else if (X > 0xFFFF)\
50  I = NTREEBINS-1;\
51  else {\
52  unsigned int K = (unsigned) sizeof(X)*__CHAR_BIT__ - 1 - (unsigned) __builtin_clz(X); \
53  I = (BinIndexType)((K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1)));\
54  }\
55 }
56 
57 #elif defined (__INTEL_COMPILER)
58 #define cookmem_getLargeBinIndex(S, I)\
59 {\
60  std::size_t X = S >> TREEBIN_SHIFT;\
61  if (X == 0)\
62  I = 0;\
63  else if (X > 0xFFFF)\
64  I = NTREEBINS-1;\
65  else {\
66  unsigned int K = _bit_scan_reverse (X); \
67  I = (BinIndexType)((K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1)));\
68  }\
69 }
70 
71 #elif defined(_MSC_VER) && _MSC_VER>=1300
72 #define cookmem_getLargeBinIndex(S, I)\
73 {\
74  std::size_t X = S >> TREEBIN_SHIFT;\
75  if (X == 0)\
76  I = 0;\
77  else if (X > 0xFFFF)\
78  I = NTREEBINS-1;\
79  else {\
80  unsigned int K;\
81  _BitScanReverse((DWORD *) &K, (DWORD) X);\
82  I = (BinIndexType)((K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1)));\
83  }\
84 }
85 
86 #else /* GNUC */
87 #define cookmem_getLargeBinIndex(S, I)\
88 {\
89  std::size_t X = S >> TREEBIN_SHIFT;\
90  if (X == 0)\
91  I = 0;\
92  else if (X > 0xFFFF)\
93  I = NTREEBINS-1;\
94  else {\
95  unsigned int Y = (unsigned int)X;\
96  unsigned int N = ((Y - 0x100) >> 16) & 8;\
97  unsigned int K = (((Y <<= N) - 0x1000) >> 16) & 4;\
98  N += K;\
99  N += K = (((Y <<= K) - 0x4000) >> 16) & 2;\
100  K = 14 - N + ((Y <<= K) >> 15);\
101  I = (K << 1) + ((S >> (K + (TREEBIN_SHIFT-1)) & 1));\
102  }\
103 }
104 #endif /* GNUC */
105 
106 #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
107 #define cookmem_bit2treeIndex(X, I)\
108 {\
109  unsigned int J;\
110  J = __builtin_ctz(X); \
111  I = (BinIndexType)J;\
112 }
113 
114 #elif defined (__INTEL_COMPILER)
115 #define cookmem_bit2treeIndex(X, I)\
116 {\
117  unsigned int J;\
118  J = _bit_scan_forward (X); \
119  I = (BinIndexType)J;\
120 }
121 
122 #elif defined(_MSC_VER) && _MSC_VER>=1300
123 #define cookmem_bit2treeIndex(X, I)\
124 {\
125  unsigned int J;\
126  _BitScanForward((DWORD *) &J, X);\
127  I = (BinIndexType)J;\
128 }
129 
130 #elif USE_BUILTIN_FFS
131 #define cookmem_bit2treeIndex(X, I) I = ffs(X)-1
132 
133 #else
134 #define cookmem_bit2treeIndex(X, I)\
135 {\
136  unsigned int Y = X - 1;\
137  unsigned int K = Y >> (16-4) & 16;\
138  unsigned int N = K; Y >>= K;\
139  N += K = Y >> (8-3) & 8; Y >>= K;\
140  N += K = Y >> (4-2) & 4; Y >>= K;\
141  N += K = Y >> (2-1) & 2; Y >>= K;\
142  N += K = Y >> (1-0) & 1; Y >>= K;\
143  I = (BinIndexType)(N + Y);\
144 }
145 #endif /* GNUC */

406  T* allocate(std::size_t size)
407  {
408  std::size_t chunkSize;
409 
410  if (size < MIN_LARGE_REQUEST)
411  {
412  BinIndexType binIndex;
413  BinIndexType smallBits;
414  chunkSize = (size < MIN_REQUEST) ? MIN_CHUNK_SIZE : calcChunkSize(size);
415  binIndex = getSmallBinIndex(chunkSize);
416  smallBits = m_smallMap >> binIndex;
417 
418  /*
419  * Check if two smallest chunk lists that can satisfy the request.
420  * Note that since the chunk is quite tight, it is not necessary
421  * to split the chunk since there will not be enough rooms.
422  */
423  if ((smallBits & 0x3U) != 0)
424  {
425  MemChunk* chunk;
426  binIndex += ~smallBits & 1; /* Uses next bin if idx empty */
427 
428  chunk = removeSmallChunkList(binIndex);
429  COOKMEM_ASSERT(chunk != nullptr && chunk->getSize() == getSmallBinSize (binIndex));
430 
431  // mark that this chunk is no longer free.
432  chunk->setFootSize(0);
433 
434  return chunk2Mem(chunk);
435  }
436  else
437  {
438  if (smallBits != 0)
439  { /* Use chunk in next nonempty smallbin */
440  MemChunk* chunk;
441  BinIndexType i;
442  BinIndexType leftbits = (smallBits << binIndex) & left_bits(idx2bit(binIndex));
443  BinIndexType leastbit = least_bit(leftbits);
444  cookmem_bit2treeIndex(leastbit, i);
445 
446  chunk = removeSmallChunkList(i);
447  COOKMEM_ASSERT(chunk != nullptr && chunk->getSize() == getSmallBinSize(i));
448 
449  chunk = splitChunk (chunk, chunkSize);
450 
451  return chunk2Mem(chunk);
452  }
453  }
454  }
455  else if (size >= MAX_REQUEST)
456  {
457  // The request size is too big.
458  return nullptr;
459  }
460  else
461  {
462  chunkSize = calcChunkSize(size);
463  }
464 
465  MemChunk* chunk;
466 
467  /*
468  * Check if we have enough space in existing chunks.
469  */
470  if (m_treeMap != 0)
471  {
472  chunk = treeMalloc (chunkSize);
473  }
474  else
475  {
476  chunk = nullptr;
477  }
478 
479  /*
480  * At this point, we have to allocate memory from arena.
481  */
482  if (chunk == nullptr)
483  {
484  chunk = arenaAlloc(chunkSize);
485 
486  if (chunk == nullptr)
487  {
488  return nullptr;
489  }
490  }
491 
492  return chunk2Mem(chunk);
493  }

504  T* reallocate(T* ptr, std::size_t size)
505  {
506  if (ptr == nullptr)
507  {
508  return allocate(size);
509  }
510 
511  MemChunk* chunk = mem2Chunk (ptr);
512  std::size_t oldChunkSize = chunk->getSize ();
513 
514  std::size_t newChunkSize = (size < MIN_REQUEST) ? MIN_CHUNK_SIZE : calcChunkSize(size);
515 
516  if (oldChunkSize < newChunkSize)
517  {
518  // For now we use a simple algorithm that allocate new memory.
519  T* newPtr = allocate (size);
520  if (newPtr)
521  {
522  memcpy(newPtr, ptr, oldChunkSize - CHUNK_OVERHEAD);
523  deallocate (ptr);
524  }
525  return newPtr;
526  }
527  else
528  {
529  // simply split the chunk
530  splitChunk (chunk, newChunkSize);
531  return ptr;
532  }
533  }

701  inline MemChunk* treeMalloc (std::size_t size)
702  {
703  MemChunk* chunk = nullptr;
704  std::size_t actualSize = size;
705 
706  BinIndexType binIndex;
707  cookmem_getLargeBinIndex(size, binIndex);
708 
709  for (; binIndex < NTREEBINS; ++binIndex)
710  {
711  PtrAVLTree& tree = treeAt(binIndex);
712  if (!tree.isEmpty ())
713  {
714  chunk = reinterpret_cast<MemChunk*>(tree.remove(actualSize));
715  if (chunk != nullptr)
716  break;
717  }
718  }
719 
720  if (chunk != nullptr)
721  {
722  return splitChunk (chunk, size);
723  }
724  return nullptr;
725  }

737  inline MemChunk* splitChunk (MemChunk* chunk, std::size_t size)
738  {
739  // the size of the 2nd half after the split
740  std::size_t remainSize = chunk->getSize() - size;
741 
742  if (remainSize >= MIN_CHUNK_SIZE)
743  {
744  MemChunk* remainChunk = (MemChunk*)((char*)chunk + size);
745 
746  chunk->setSize(size);
747  chunk->setFootSize(0);
748 
749  remainChunk->setSize(remainSize);
750  remainChunk->setFootSize(remainSize);
751 
752  addChunk(remainChunk);
753  }
754  return chunk;
755  }

765  MemChunk* arenaAlloc(std::size_t chunkSize)
766  {
767  std::size_t estSize; /* allocation size */
768 
769  estSize = chunkSize + SEGMENT_OVERHEAD;
770  if (estSize <= chunkSize)
771  {
772  return nullptr; /* wraparound */
773  }
774 
775  /*
776  * Check user specified foot print limit
777  */
778  if (m_footprintLimit != 0)
779  {
780  std::size_t fp = m_footprint + estSize;
781  if (fp <= m_footprint || fp > m_footprintLimit)
782  return nullptr;
783  }
784 
785  /*
786  * Request memory segments from arena.
787  */
788  std::size_t segSize = estSize;
789  MemSegment* seg = (MemSegment*)m_arena.getSegment(segSize);
790 
791  /*
792  * Initialize the segment obtained.
793  */
794  if (seg)
795  {
796 
797  if ((m_footprint += segSize) > m_maxFootprint)
798  m_maxFootprint = m_footprint;
799 
800  MemChunk* chunk = seg->init(segSize);
801 
802  if (m_segList == nullptr)
803  {
804  m_segList = seg;
805  seg->next = nullptr;
806 
807  m_release_checks = MAX_RELEASE_CHECK_RATE;
808  }
809  else
810  {
811  seg->next = m_segList;
812  m_segList = seg;
813  }
814 
815  return splitChunk(chunk, chunkSize);
816  }
817  return nullptr;
818  }
